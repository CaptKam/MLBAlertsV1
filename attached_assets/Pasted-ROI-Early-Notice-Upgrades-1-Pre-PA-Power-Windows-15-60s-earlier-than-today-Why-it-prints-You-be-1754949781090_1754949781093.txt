on_deck_power_spike:
  gate: state == on_deck
  conditions:
    - PowerScore >= 65   # or >=55 for B-tier variant
    - park_hr_factor >= 1.00
    - wind_out_mph >= 4 or temp_f >= 80
  dedup_window: 60s
  priority: High

impending_platoon:
  gate: state in [on_deck, at_bat]
  conditions:
    - platoon_iso_delta >= 0.040
    - (tto >= 3) or (weak_same_hand_reliever_warming == true)
  dedup_window: 60s
  priority: High

noncomp_cluster:
  gate: state == at_bat
  conditions:
    - noncomp_pitches_last4 >= 3  # balls or 12+ inches off target
    - (batter_power_tier in [A,B])
  dedup_window: 30s
  priority: High

behind_count_power:
  gate: state == at_bat
  conditions:
    - count in ["2-0","3-1"]
    - (pitcher_bb_pct >= p70) or (csw_pct <= league_avg)
  dedup_window: 30s
  priority: High

impending_hook:
  gate: half_inning_active == true
  conditions:
    - tto >= 3
    - pitch_count >= 90
    - bullpen_hot == true
  dedup_window: 120s
  priority: High

reliever_first_batter_risk:
  gate: pitching_change == true
  conditions:
    - (reliever_reverse_splits == true) or (reliever_back_to_back_days == true)
  dedup_window: 120s
  priority: Medium-High

gust_spike:
  gate: between_innings or batter_on_deck == true
  conditions:
    - delta_wind_out_mph_3min >= 5
    - park_hr_factor >= 1.05
    - next_due_power_tier in [A,B]
  dedup_window: 5m
  priority: High
