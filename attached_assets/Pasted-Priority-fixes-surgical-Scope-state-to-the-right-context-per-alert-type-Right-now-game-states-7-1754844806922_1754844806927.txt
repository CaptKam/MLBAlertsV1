def _half_inning_key(game):
    # top/bottom + inning number from feed
    return f"{game['about']['isTopInning']}_{game['about']['inning']}"

def _plate_appearance_id(play):
    # prefer official atBatIndex if available
    return play.get("about", {}).get("atBatIndex")

def _bases_hash(runners):
    # stable order
    return "_".join(sorted(runners)) if runners else "EMPTY"

def _make_keys(game_id, alert_type, data):
    cfg = ALERT_CONFIG[alert_type]
    half_inning = data.get("half_inning") or _half_inning_key(data["context_game"])
    pa_id = data.get("pa_id") or _plate_appearance_id(data.get("context_play", {}))

    # content per config
    content_pieces = []
    for f in cfg["content_fields"]:
        if f == "bases_hash":
            content_pieces.append(_bases_hash(data.get("runners")))
        elif f == "pa_id":
            content_pieces.append(str(pa_id))
        else:
            content_pieces.append(str(data.get(f)))
    content = "|".join(content_pieces)

    # scope → simple key
    if cfg["scope"] == "play":
        simple = f"{game_id}_{alert_type}_{data.get('play_id')}"
    elif cfg["scope"] == "plate_appearance":
        simple = f"{game_id}_{alert_type}_{half_inning}_{pa_id}"
    elif cfg["scope"] == "half_inning":
        simple = f"{game_id}_{alert_type}_{half_inning}"
    else:
        simple = f"{game_id}_{alert_type}"

    dedup = f"{game_id}_{alert_type}_{content}"
    return simple, dedup, cfg

def _is_new_alert(self, game_id: int, alert_type: str, data) -> bool:
    simple_key, dedup_key, cfg = _make_keys(game_id, alert_type, data)
    now = time.time()

    # time-based throttle
    last_ts = self.recent_alerts.get(dedup_key)
    if last_ts and (now - last_ts) < cfg["window"]:
        return False

    # state-based gate with scoped state
    prev_state = self.game_states.get(simple_key)
    if prev_state == data.get("state_value"):  # you set state_value per alert usage
        # consider re‑alert if configured (persistent condition)
        ra = cfg.get("realert_after_secs")
        if ra:
            last_sent = self.last_sent_by_simple.get(simple_key, 0)
            if (now - last_sent) >= ra:
                return True  # allow periodic reminder
        return False

    return True

def _add_alert(..., alert_type: str, data: dict):
    simple_key, dedup_key, cfg = _make_keys(game_id, alert_type, data)
    now = time.time()
    self.recent_alerts[dedup_key] = now
    # define what “state_value” means per alert before calling _is_new_alert
    self.game_states[simple_key] = data.get("state_value")
    self.last_sent_by_simple[simple_key] = now
